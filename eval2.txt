todo by express

const express =require("express");
const app=express()
app.use(express.json())
const todos=[];
app.get("/todos",(req,res)=>{
    return res.send(
        todos
    )
})

app.post("/todo",(req,res)=>{
    try{
    const {todo}=req.body;
    const {task}=todo;
    // console.log(task)

    let max=0;
    todos.forEach(todo=>{
        max=Math.max(max,todo.id)
    })

     let data={
        id:max+1,
        task,
        createdAt:new Date()
     }
     todos.push(data)
    return res.send("todo has been added")
    }catch(ex){
        console.error(ex);
        return res.status(500).send("Internal Server Error");
    }
})

app.put("/todo/id",(req,res)=>{
    // console.log(req.body)
   let temp=null
   let changeTodo=todos.find((e,index)=>{
    if(e.id==req.params.id){
        temp=index
        return e.id==req.params.id
    }
   })
   changeTodo={...changeTodo,...req.body}
   todos[temp]=changeTodo
   res.send(`put:${json.stringify(todos)}`)
})

 file uploader

 const express=require('express')
const multer=require('multer')
const app=express()
//the upload middleware
const upload=multer({
    storage:multer.diskStorage({
        destination:function(req,res,cb){
            cb(null,"./uploads")
        },
        filename:function(req,file,cb){
            cb(null,file.filename+".jpg")
        }
    })
})
app.post('/upload',upload.single("image"),(req,res)=>{
// console.log(req.file)
res.send(" file uploaded")
})
app.listen(8000,()=>{
    console.log("your server is running on port 8000")
})
